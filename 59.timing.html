Timing Events

The window object allows execution of code at specified time intervals. These timing intervals are called timing events.

The two key methods to use with JavaScript are:

  • setTimeout(function, milliseconds)
  • setInterval(function, milliseconds)

The setTimeout() and setInterval() are both methods of the HTML DOM Window object.

Timed Alert

Set a timeout for an alert (3 seconds):


			function myFunction1() {
				alert("Hello There!");
			}

Timed Text

Display the number of seconds past at 2, 4, and 6 seconds:


			function dispSeconds() {
				setTimeout(myTime1, 2000)
				setTimeout(myTime2, 4000)
				setTimeout(myTime3, 6000)
			}

			function myTime1() {
				document.getElementById("demo2").innerHTML = "2 seconds";
			}
			function myTime2() {
				document.getElementById("demo2").innerHTML = "4 seconds";
			}
			function myTime3() {
				document.getElementById("demo2").innerHTML = "6 seconds";
			}

Continuous Timer

Display a timer that runs "forever":


			var s = 0;
			function myCounter() {
				document.getElementById("demo3").innerHTML = ++s;
			}

Continuous Timer with Stop Button

Display a timer with a start and stop button:


			var c = 0;
			function myCounter2() {
				document.getElementById("demo4").innerHTML = ++c;
			}

Clock

Create a clock using a timing event:


			function startTime() {
				var today = new Date();
				var h = today.getHours();
				var m = today.getMinutes();
				var s = today.getSeconds();
				m = checkTime(m);
				s = checkTime(s);

			document.getElementById("demo5").innerHTML =
			h + ":" + m + ":" + s;

			var t = setTimeout(startTime, 500);
			}

			function checkTime(i) {
				//add zeros in front of numbers less than 10
				if (i < 10) {
					i = "0" + i
				};
					return i;
			}
