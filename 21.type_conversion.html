Convert Number to String (toString)

Convert a number to a string using toString():


			var x = 234;
			document.getElementById("demo3").innerHTML = x.toString();

Convert Number to String (String)

Convert a number to a string using String():


			var x = 345;
			document.getElementById("demo4").innerHTML = String(x);

Determine if is Array

Determine if a variable is an array using isArray():


			var colors = ["Red", "Green", "Blue", "Orange"];
			document.getElementById("demo5").innerHTML =
			isArray(colors);

			function isArray(myArray) {
				return myArray.constructor.toString().indexOf("Array") > -1;
			}

Determine if is a Date

Determine if a variable is a date using isDate():


			var myDate = new Date();
			document.getElementById("demo6").innerHTML = isDate(myDate);

			function isDate(myDate) {
				return myDate.constructor.toString().indexOf("Date") > -1;
			}
