JavaScript Loops

Loops can execute a block of code a number of times.

For Loop

Click the button to loop through the block of code 5 times:


			function myFunction1() {
				var x = "", i;
				for (i = 0; i < 5; i++) {
					x = x + "The number is " + i + "
";
				}
				document.getElementById("demo1").innerHTML = x;
			}

					"";

While Loop

Loop through code while i is less than 10:


			function myFunction3() {
				var text = "";
				var i = 0;
				while (i < 10) {
					text += "
 The number is " + i;
					i++;
				}
				document.getElementById("demo3").innerHTML = text;
			}

Do While Loop

Execute code (do) as long as a certain condition is true (while):


			function myFunction4() {
				var text = "",
				i = 0;

				do {
					text += "
The number is " + i;
					i++;
				}
				while (i < 10);
				document.getElementById("demo4").innerHTML = text;
			}

Break a Loop

Stop the function's execution with a break in the loop:


			function myFunction5() {
				var text = "", i;
				for (i = 0; i < 10; i++) {
					if (i === 3) { break; }
					text += "The number is " + i + "
";
				}
				document.getElementById("demo5").innerHTML = text;
			}

Continue

Break the loop (breaks at 3) and then continue execution of loop:


			function myFunction6() {
				var text, i;
				for (i = 0; i < 10; i++) {
					if (i === 3) {continue;}
					text += "The number is " + i + "
";
				}
				document.getElementById("demo6").innerHTML = text;
			}

for in

Use the for...in statement to loop through the properties of an object:


			function myFunction7() {
				var txt = "", x;
				var person = {
					fname: "John",
					lname: "Doe",
					age: 49
				};
				for (x in person) {
					txt += person[x] + " ";
				}
				document.getElementById("demo7").innerHTML = txt;
			}
