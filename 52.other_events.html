onkeyup (display key code)

Click on the screen and then press a keyboard key to get the keycode:


			//onkeyup event attached to body element
			function keyCode(event) {
				document.getElementById("demo1").innerHTML = event.keyCode;
			}

onmousedown (is key pressed?)

Display an alert that tells you if the shift key was pressed or not:


			//onmousedown attached to body element
			function isKeyPressed(event) {
				var txt = "The shift key was NOT pressed!";
				if (event.shiftKey == 1) {
					txt = "The shift key WAS pressed!";
				}
				document.getElementById("demo4").innerHTML = txt;
			}
