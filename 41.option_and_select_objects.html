Option and Select Objects

Enable/Disable Dropdown

Enable and disable a dropdown list:



			function disable() {
				var x = document.getElementById("select1");
				if (x.disabled === true) {
					x.disabled = false;
				} else {
					x.disabled = true;
				}
			}

Get ID

Get the id of the form that contains the dropdown list:



			function getId() {
				var x = document.getElementById("select2").form.id;
				document.getElementById("demo2").innerHTML = x;
			}

Options

Get the number of options in the dropdown list:



			function getOptions() {
				document.getElementById("demo3").innerHTML =
				document.getElementById("select3").length;
			}

Multiline Dropdown

Make the dropdown list multiline by changing its size:



			function changeSize() {
				document.getElementById("select4").size = 3;
			}

Multi-Select

Select multiple options in a dropdown list using the "shift" key:



			function multiSelect() {
				document.getElementById("select5").multiple = true;
			}

Display Selected

Display the selected options in a dropdown list:



			function displaySelect() {
				var obj = document.getElementById("select6");
				document.getElementById("demo6").innerHTML = obj.options[obj.selectedIndex].text;
			}

Display All Options

Display all options in a dropdown list:



			function displayOptions() {
				var x = document.getElementById("select7");
				var txt = ""
				var i;
				for (i = 0; i < x.length; i++) {
					txt += x.options[i].text + " ";
				}
				document.getElementById("demo7").innerHTML = txt;
			}

Index

Display the index of a selected option in a dropdown list:



			function displayIndex() {
				var x = document.getElementById("select8").selectedIndex;
				document.getElementById("demo8").innerHTML = x;
			}

Change Text

Change the text of the selected option:



			function changeText() {
				var x = document.getElementById("select9");
				x.options[x.selectedIndex].text = "Brown";
			}

Remove Options

Remove options from a dropdown list:



			function removeOption() {
				var x = document.getElementById("select10");
				x.remove(x.selectedIndex);
			}
