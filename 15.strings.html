JavaScript Strings

JavaScript strings are used for storing and manipulating text.

A JavaScript string simply stores a series of characters like "John Doe".

A string can be any text inside quotes. You can use single or double quotes.

String With Single or Double Quotes


			var firstName = "Joey", lastName = 'Atwood';

			document.getElementById("demo1").innerHTML = firstName + " " + lastName;

String Examples:


			var firstName = "Joey",
			lastName = 'Atwood',
			car = "Audi",
			favoriteQuote = "'To thine own self, be true!'";

			document.getElementById("demo2").innerHTML =
			firstName +
			lastName +
			car +
			favoriteQuote;

			//add line breaks

Backslash Before Quotes Accepts Quotes as Quotes


			var x = 'It\'s alright';
			var y = "We are the so called \'Champions\.'";

			document.getElementById("demo3").innerHTML = x + " " + y;

Find the Length of a String


			var txt = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
			document.getElementById("demo4").innerHTML = txt.length;

Break Text String with a Backslash


			document.getElementById("demo5").innerHTML =
			"Hello \ there!";

You Cannot Break a Codeline with a Backslash

You can't put a backslash in a line of code! (it will not display)


			document.getElementById("demo6").innerHTML = \
			"Hello There!";

Find Position of First Occurrence of Text in a String (.indexOf())

Find where the word text is in this sentence by using indexOf()


			function findText() {
				var string = document.getElementById("p1").innerHTML;
				var position = string.indexOf("text");

				document.getElementById("demo7").innerHTML = position;
			}

Search for Text in a String and Return Text if Found (.match())


			function findString() {
				var string = "The rain in Spain stays mainly in the plain";

				var found = string.match(/ain/g);

				document.getElementById("demo8").innerHTML = found;
			}

Replace Characters in a String (.replace())

Please visit Google!


			function replace() {
				var str = document.getElementById("demo9").innerHTML;
				var txt = str.replace("Google", "Bing");

				document.getElementById("demo9").innerHTML = txt;
			}

Convert String to Upper Case (.toUpperCase())

This is really, really cool!


			function toUpper() {
				var txt = document.getElementById("demo10").innerHTML;

				document.getElementById("demo10").innerHTML = txt.toUpperCase();
			}

Convert String to Lower Case (.toLowerCase())

THIS IS ALSO REALLY, REALLY COOL!

Split a String into an Array (.split())


			function splitString() {
				var str = "a,b,c,d,e,f,g";
				var arr = str.split(",");

				document.getElementById("demo12").innerHTML = arr[2];
			}
