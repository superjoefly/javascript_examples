Window Object

The window object is supported by all browsers; it represents the browser's window.

All global JavaScript objects, functions, and variables become members of the window object.

window.open

Open a new window when a button is clicked:


			function openWindow() {
				window.open();
			}

window.open (with specs)


			function specWindow() {
				window.open("https://www.bing.com", "_blank", "toolbar=yes, location=yes, directories=no, status=no, menubar=yes, scrollbars=yes, resizable=no, copyhistory=yes, width=400, height=400");
			}

window.open (blur/focus)


			var myWindow;
			function newWindow() {
				myWindow = window.open("", "", "width=400, height=400");
				myWindow.blur();
			}

			function windowBlur() {
				myWindow.blur();
			}
			function windowFocus() {
				myWindow.focus();
			}

window.close

Close the new window:


			function newWindow2() {
				myWindow = window.open("", "", "width=400, height=400");
			}

			function closeWindow() {
				myWindow.close();
			}

Is New Window Open?

Check if new window is open or not:


			var myWindow;
			function openWindow3() {
				myWindow = window.open("", "", "width=400, height=400");
			}

			function closeWindow3() {
				if (myWindow) {
					myWindow.close();
				}
			}

			function checkWindow3() {
				var msg = "";
				if (!myWindow) {
					msg = "Window never opened";
				} else {
					if (myWindow.closed) {
						msg = "Window closed";
					} else {
						msg = "Window opened";
					}
				}
				document.getElementById("demo3").innerHTML = msg;
			}

window.open (write to parent window)


			function openWindow4() {
				var myWindow = window.open("", "", "width=400, height=400");

				myWindow.opener.document.getElementById("demo4").innerHTML =
				"A new window was opened!";
			}

.moveBy

Move the new window relative to its current position:


			var myWindow;
			function openWindow5() {
				myWindow = window.open("", "", "width=400, height=400");
			}
			function moveWindow5() {
				myWindow.moveBy(300, 300);
			}

.moveTo

Move the new window to a specified position:


			var myWindow;
			function openWindow6() {
				myWindow = window.open("", "", "width=400, height=400");
			}
			function moveWindow6() {
				myWindow.moveTo(300, 0);
				myWindow.focus();
			}

window.print

Print the current page:


			function printPage() {
				window.print();
			}

.resize

Resize a new page by a specified number of pixels:


			var w;
			function openWindow7() {
				w = window.open("", "", "width=400, height=400");
				w.focus();
			}
			function resizeWindow7() {
				w.resizeBy(50, 50);
				w.focus();
			}

.resizeTo

Resize a new window to a specified size:


			var w;
			function openWindow8() {
				w = window.open("", "", "width=400, height=400");
				w.focus();
			}
			function resizeWindow8() {
				w.resizeTo(500, 500);
				w.focus();
			}

.scrollBy

Scroll the document by a specified number of pixels:


			function scrollWindow() {
				window.scrollBy(0, 100);
			}

.scrollTo

Scroll the document to a specified position:


			function scrollWindow2() {
				window.scrollTo(0, 100);
			}
