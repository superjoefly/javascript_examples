Change Border

Change the width of a table's border:


			function myFunction1() {
			document.getElementById("myTable").style.border="5px solid"
			}

Inner HTML

Find the inner HTML of a cell


			function myFunction2() {
				var x = document.getElementById("myTable").rows[0].cells;
				document.getElementById("demo2").innerHTML = x[0].innerHTML;
			}

Add Rows

Add rows to a table:


			function insRow(id) {
				var x = document.getElementById(id).insertRow(1);
				var y = x.insertCell(0);
				var z = x.insertCell(1);
				y.innerHTML = z.innerHTML = "New";
			}

Change Content

Change the content of a table cell (see result above):


			function changeContent(id, row, cell, content) {
			    var x = document.getElementById(id).rows[row].cells;
			    x[cell].innerHTML = content;
			}
