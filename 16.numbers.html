Numbers

JavaScript has only one type of number. Numbers can be written with or without decimals. A number inside of quotes is a string.

Number Accuracy

Integers are considered acurate only up to 15 digits.


			function myFunction() {
				var x = 999999999999999;
				var y = 9999999999999999;

			document.getElementById("demo1").innerHTML =
			x + y;
		}

		//add line break

Floating Point Arithmetic

Floating point arithmetic is not always 100% accurate.


			function myFunction2() {
				var x = .2 + .1;

				document.getElementById("demo2").innerHTML =
				".2 + .1 = " + x;
			}

Gain Accuracy

Accuracy can be gained by first multiplying and then dividing by 10.


			function myFunction3() {
				var x = (.2 * 10 + .1 * 10) / 10;
				document.getElementById("demo3").innerHTML =
				".2 + .1 = " + x;
			}

Constants

Constants preceded by 0x, are interpreted as hexidecimal.


			function myFunction4() {
				document.getElementById("demo4").innerHTML =
				"0xFF = " + 0xFF;
			}

base 16, base 8, base 2

The toString() can output numbers as base 16 (hex), base 8 (octal), or base 2 (binary).


			function myFunction5() {
				var myNumber = 536;

				document.getElementById("demo5").innerHTML =
				"536 = " +
				myNumber + " Decimal, " +
				myNumber.toString(16) + " Hexadecimal, " +
				myNumber.toString(8) + " Octal, " +
				myNumber.toString(2) + " Binary."
			}

Infinity

JavaScript will generate Infinity if you calculate a too large number.


			function myFunction6() {
				var myNumber = 2;
				var txt = "";
				while (myNumber != Infinity) {
					myNumber = myNumber * myNumber;
					txt = txt + myNumber //add line break;
				}

				document.getElementById("demo6").innerHTML = txt;
			}

Division by Zero

Division by 0 also generates infinity.


			function myFunction7() {
				var x = 2/0;
				var y = -2/0;

				document.getElementById("demo7").innerHTML =
				x + y;
			}

			//add line break

Number Divided by a String

A number divided by a string is not a number (NaN).

A number divided by a numeric string IS a number.


			function myFunction8() {
				var x = 1000/ "something";
				var y = 1000 / "10";

				document.getElementById("demo8").innerHTML =
				x + y;
			}

			//add line break
