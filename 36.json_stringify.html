JSON Stringify

A common use of JSON is to exchange data to/from a webserver.

When sending data to a webserver, the data has to be a string.

You can convert a JavaScript object into a string with JSON.stringify().

Stringify Object

Create a JSON string from a JavaScript object:


			var obj = {
				"name":"Jeremy",
				"age": 22,
				"city": "New Jersey"
			};
			var myJSON = JSON.stringify(obj);
			document.getElementById("demo1").innerHTML = myJSON;

Stringify Array

Create JSON string from a JavaScript array:


			var arr = [
				"Jeremy", "Billybob", "Sallydoo", "Joeblow"
			];
			var myJSON = JSON.stringify(arr);
			document.getElementById("demo2").innerHTML = myJSON;

Stringify Dates

Convert any date object into a JSON string:


			var obj = {
				"name":"Frank",
				"age": 44,
				"today": new Date(),
				"city": "New York"
			};
			var myJSON = JSON.stringify(obj);
			document.getElementById("demo3").innerHTML = myJSON;

Stringify Functions

JSON.stringify will remove functions from an object:


			var obj = {
				"name": "Jill",
				"age": function() {return 45;},
				"city": "Chicago"
			};
			var myJSON = JSON.stringify(obj);
			document.getElementById("demo4").innerHTML = myJSON;

Stringify Functions (toString)

To keep the function in the JSON object, first convert it to a string using the toString() method:


			var obj = {
				"name": "Harry",
				"age":function() {return 54;},
				"city":"Montgomery"
			};
			obj.age = obj.age.toString();
			var myJSON = JSON.stringify(obj);
			document.getElementById("demo5").innerHTML = myJSON;
