JSON Arrays

Arrays in JSON are almost the same as arrays in JavaScript, however, in JSON, arrays must be of type string, number, object, boolean or null.

Array Values

Access JSON array values:


			var myObj, x;
			myObj = {
				"name": "Sally",
				"age": 33,
				"cars": ["VW", "Ford", "BMW"]
			};
			x = myObj.cars[0];
			document.getElementById("demo1").innerHTML = x;

Looping with for...in

Loop through JSON array using for...in:


			var myObj, i, x = "";
			myObj = {
				"name":"Joe",
				"age": 37,
				"cars": [
					"Audi", "Fiat", "Cadillac"
				]
			};
			for (i in myObj.cars) {
				x += myObj.cars[i] + " ";
			}
			document.getElementById("demo2").innerHTML = x;

Looping using for

Loop through a JSON array using for:


			var myObj, i, x = "";
			myObj = {
				"name": "Bob",
				"age": 45,
				"cars": [
					"Fiat", "Ford", "Subaru"
				]
			};

			for (i = 0; i < myObj.cars.length; i++) {
				x += myObj.cars[i] + " ";
			}

			document.getElementById("demo3").innerHTML = x;

Access Nested Arrays

Loop through an array inside of an array:


			var myObj, i, j, x = "";

			myObj = {
				"name": "Joe",
				"age": "37",
				"cars": [
					{
						"name":"Toyota", "models": ["Tacoma", "Carolla", "Camry"]
					},
					{
						"name":"Ford", "models": ["Focus", "Taurus", "Mustang"]
					},
					{
						"name":"Fiat", "models": ["Panda", "500"]
					}
				]
			}

			for (i in myObj.cars) {
				x += "
" + myObj.cars[i].name + "
";
				for (j in myObj.cars[i].models) {
					x += myObj.cars[i].models[j] + " ";
				}
			}

			document.getElementById("demo4").innerHTML = x;

Modify Array Values

Modify an array value of a JSON object:


			var myObj, i, x = "";
			myObj = {
				"name":"Jessy",
				"age": 29,
				"cars": [
					"Toyota", "BMW", "Audi"
				]
			};
			myObj.cars[1] = "Ford";

			for (i in myObj.cars) {
				x += myObj.cars[i] + " ";
			}

			document.getElementById("demo5").innerHTML = x;

Delete Array Items


			var myObj, i, x = "";
			myObj = {
				"name":"Hank",
				"age":47,
				"cars": [
					"Ford", "Buick", "Fiat", "Toyota"
				]
			};
			delete myObj.cars[1];

			for (i in myObj.cars) {
				x += myObj.cars[i] + " ";
			}

			document.getElementById("demo6").innerHTML = x;
