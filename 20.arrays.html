JavaScript Arrays

JavaScript arrays are used to store multiple values in a single variable.

Arrays

Create an array of names:


			var names = [
				"Bill",
				"Jack",
				"David"
			];

			document.getElementById("demo1").innerHTML = names;

concat()

Join two arrays using concat():


			var girls = ["Jane", "Jill", "Jessica"];
			var boys = ["Jack", "David", "Brian"];
			var friends = girls.concat(boys);

			document.getElementById("demo2").innerHTML = friends;

concat() - more than two

Join more than two arrays:


			var arr1 = ["Jack", "John", "Jerry"];
			var arr2 = ["Ben", "Bill", "Bob"];
			var arr3 = ["Jessical", "Jill", "Jane"];

			var friends = arr1.concat(arr2, arr3);

			document.getElementById("demo3").innerHTML = friends;

join()

Join array elements into a string using join():


			var colors = ["red", "orange", "yellow", "green", "blue", "violet"];

			document.getElementById("demo4").innerHTML =

slice()

When the slice method is given two arguments, it selects array elements from the start argument up to (but not including) the end argument.

Select elements from an array using slice():


			var colors = ["red", "orange", "yellow", "green", "blue", "violet"];

			document.getElementById("demo9").innerHTML = colors;

			function myFunction5() {
			var favoriteColors = colors.slice(1,3);

			document.getElementById("demo9").innerHTML =
			favoriteColors;
			}

sort() - objects by number

Sort objects by numberic properties usng sort():


			var cars = [
				{name:"Buick", year:2010},
				{name:"Honda", year:1989},
				{name:"Toyota", year:1950}
			]

			displayCars();

			function myFunction11() {
				cars.sort(function(a, b) {
					return a.year - b.year
				});
				displayCars();
			}

			function displayCars() {
				document.getElementById("demo15").innerHTML =
				cars[0].name + " " + cars[0].year + "
" +
				cars[1].name + " " + cars[1].year + "
" +
				cars[2].name + " " + cars[2].year;
			}

sort() - objects by string

Sort objects by string properties using sort():


			var cars = [
				{name:"Honda", year:2010},
				{name:"Buick", year:1989},
				{name:"Toyota", year:1950}
			]

			displayCars();

			function myFunction12() {
				cars.sort(function(a, b) {
					var x = a.name.toLowerCase();
					var y = b.name.toLowerCase();
					if (x < y) {return -1;}
					if (x > y) {return 1;}
					return 0;
				});
				displayCars();
			}

			function displayCars() {
			  document.getElementById("demo16").innerHTML =
			  cars[0].name + " " + cars[0].year + "
" +
			  cars[1].name + " " + cars[1].year + "
" +
			  cars[2].name + " " + cars[2].year;
			}
