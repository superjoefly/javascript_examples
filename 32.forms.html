JavaScript Form Validation

HTML form validation can be done by JavaScript:

If a form field is (fname) is empty, alert a message and return false to prevent the form from being submitted:

Name: 

			function validateForm() {
				var x = document.forms["myForm"] ["fname"].value;
				if (x == "") {
					alert("Name must be filled in!");
					return false;
				}
			}

Validate Numeric Input

JavaScript can validate numberic input:

Please input a number between 1 and 10:


			function myFunction() {
				var x, text;
				x = document.getElementById("numb").value;
				if (isNaN(x) || x < 1 || x > 10) {
					text = "Input not valid!";
				} else {
					text = "Input okay!";
				}

				document.getElementById("demo1").innerHTML = text;
			}

checkValidity() Method

Display a message if an input field contains invalid data:

If the number is less than 100 or greater than 300, an error message will be displayed.

Enter a number and then click OK:


			function myFunction2() {
				var inpObj = document.getElementById("id1");
				if (inpObj.checkValidity() == false) {
					document.getElementById("demo2").innerHTML = inpObj.validationMessage;
				}	else {
					document.getElementById("demo2").innerHTML = "Input Okay!";
				}
			}

rangeOverflow Property

If the number in the input field is greater than 100 (the input's max attribute), display a message:

Enter a number and click OK:


			function myFunction3() {
				var txt = "";
				if (document.getElementById("id2").validity.rangeOverflow) {
					txt = "Value too large!";
				}	else {
					txt = "Input okay!";
				}
				document.getElementById("demo3").innerHTML = txt;
			}
