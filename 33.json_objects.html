JSON Objects

JSON stands for JavaScript Object Notation

JSON is syntax for storing and exchanging data.

JSON is text, written with JavaScript object notation

Accessing JSON Object

Access a JSON object using dot notation:


			var myObj, x;
			myObj = {
				"name":"John",
				"age":45,
				"car":null
			};
			x = myObj.name;
			document.getElementById("demo1").innerHTML = x;

Access a JSON object using bracket notation:


			var myObj, x;
			myObj = {
				"name":"John",
				"age":45,
				"car":null
			};
			x = myObj["age"];
			document.getElementById("demo2").innerHTML = x;

Looping

Loop through all properties in a JSON object:


			var myObj = {
				"name": "John",
				age: 45,
				"car": null
			};
			for (x in myObj) {
				document.getElementById("demo3").innerHTML += x + " ";
			}

Loop through all values in an object:


			var x, txt = "";
			var myObj = {
				"name":"John",
				"age":45,
				"car":null
			};

			for (x in myObj) {
				txt += myObj[x] + " ";
			}

			document.getElementById("demo4").innerHTML = txt;

Nested Objects

Access nested JSON objects:


			var myObj = {
				"name": "John",
				"age": 45,
				"cars": {
					"car1": "BMW",
					"car2": "Fiat",
					"car3": "Audi"
				}
			}
			//dot notation
			document.getElementById("demo5").innerHTML += myObj.cars.car2;
			//or bracket notation
			document.getElementById("demo6").innerHTML += myObj.cars["car3"];

Modifying Values

Modify a value using the dot notation:


			var myObj, i, x = "";
			myObj = {
			  "name":"John",
			  "age":30,
			  "cars": {
			    "car1":"Ford",
			    "car2":"BMW",
			    "car3":"Fiat"
			  }
			}
			myObj.cars.car2 = "Mercedes";

			for (i in myObj.cars) {
			    x += myObj.cars[i] + " ";
			}
			document.getElementById("demo7").innerHTML = x;

Modify a value using the bracket notation:


			var myObj, i, x = "";
			myObj = {
			  "name":"John",
			  "age":30,
			  "cars": {
			    "car1":"Ford",
			    "car2":"BMW",
			    "car3":"Fiat"
			  }
			}
			myObj.cars["car2"] = "Mercedes";

			for (i in myObj.cars) {
			    x += myObj.cars[i] + " ";
			}
			document.getElementById("demo8").innerHTML = x;

Delete Object Properties

Delete properties of an existing JSON object:


			var myObj, i, x = "";
			myObj = {
				"name": "Jerry",
				"age": 67,
				"cars": {
					"car1": "Audi",
					"car2": "BMW",
					"car3": "Ford"
				}
			}
			delete myObj.cars.car2;

			for (i in myObj.cars) {
				x += myObj.cars[i] + " ";
			}

			document.getElementById("demo9").innerHTML = x;
