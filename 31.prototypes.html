Object Prototypes

Every JavaScript object has a prototype. The prototype is also an object.

All JavaScript objects inherit their properties and methods from their prototype.

Creating a Prototype

Use an object constructor function to create a prototype:


			function Friend(firstName, lastName, age, eyeColor) {
				this.firstName = firstName;
				this.lastName = lastName;
				this.age = age;
				this.eyeColor = eyeColor;
			}

			var myFriend = new Friend("Bob", "Hanks", 57, "brown");

			document.getElementById("demo1").innerHTML = myFriend.lastName;

The constructor function is the prototype for Friend objects.

It is considered good practice to name constructor functions with an upper-case letter.

Adding Properties

To add a new property to a prototype, you must add it to the constructor function:


			function Friend (firstName, lastName, age, eyeColor) {
				this.firstName = firstName;
				this.lastName = lastName;
				this.age = age;
				this.eyeColor = eyeColor;
				this.nationality = "English";
			}

			var myFriend = new Friend("Jerry", "Jackson", 48, "green");

			document.getElementById("demo2").innerHTML = myFriend.nationality;

Adding Methods

A constructor function can also define methods:


			function Friend (firstName, lastName, age, eyeColor) {
				this.firstName = firstName;
				this.lastName = lastName;
				this.age = age;
				this.eyeColor = eyeColor;
				this.fullName = function() {
					return this.firstName + " " + this.lastName
				};
			}
			var myFriend = new Friend("Jack", "Bell", 50, "blue");
			document.getElementById("demo3").innerHTML = myFriend.fullName();

Prototype Property

Add new properties to an existing prototype:


			function Friend(first, last, age, eye) {
				this.firstName = first;
				this.lastName = last;
				this.age = age;
				this.eyeColor = eye;
			}
			Friend.prototype.nationality = "English";

			var myFriend = new Friend("John", "Henry", 48, "grey");

			document.getElementById("demo4").innerHTML = myFriend.eyeColor;

Add new methods to existing prototype:


			function Friend(first, last, age, eye) {
				this.firstName = first;
				this.lastName = last;
				this.age = age;
				this.eyeColor = eye;
			}
			Friend.prototype.fullName = function() {
				return this.firstName + " " + this.lastName
			};

			var myFriend = new Friend("Hank", "Williams", 55, "brown");

			document.getElementById("demo5").innerHTML = myFriend.fullName();
